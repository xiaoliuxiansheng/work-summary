> 进行读写操作
````javascript 1.8
//封装 set  k—v
function setKey(key,value,callback){
    Redis.on("connect",()=>{
        Redis.set(key,value,callback)
    })
}
//封装get  k—v
function getKey(key,callback){
    Redis.on("connect",()=>{
        Redis.get(key,callback)
    })
}
//set设置
setKey("age","18",(err,res)=>{
    if(err){
        console.log(err)
        return
    }
    console.log(res)
})
//get获取
getKey("age",(err,res)=>{
    if(err){
        console.log(err)
        return
    }
    console.log(res)
})

````
